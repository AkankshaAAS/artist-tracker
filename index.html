Your sheet format is: Event name - pre wedding, Makeup by - Team, Location - Igatpuri, RT - 12 PM
Replace your entire Apps Script with this. It reads the label-based format (Event name -, Location -, RT -) and sends eventName, location, readyTime to the tracker.
// ═══════════════════════════════════════════════════ // ARTIST AVAILABILITY TRACKER — Apps Script // Sheet: Sales Bridal Response Sheet > Responses tab // ═══════════════════════════════════════════════════ var SS_ID = ''; // ← leave blank: script uses active spreadsheet var RESP_SHEET = 'Responses'; var AVAIL_SHEET = 'Artist Availability'; var DATES_SHEET = 'BOOKING_DATES'; // PDF URL read from Drive folder, matched by filename in column AV (index 47) var _pdfCache = null; function buildPdfCache() { if (_pdfCache) return _pdfCache; _pdfCache = {}; try { var folder = DriveApp.getFolderById('1bClP7P7OoilreV175TMqUBlhdlyH52Gc'); var files = folder.getFiles(); while (files.hasNext()) { var f = files.next(); var key = f.getName().replace(/\.pdf$/i,'').trim().toLowerCase(); _pdfCache[key] = 'https://drive.google.com/file/d/' + f.getId() + '/view'; } } catch(e) { Logger.log('PDF folder: ' + e); } return _pdfCache; } function getPdfUrl(row) { // col AW (index 48) holds the PDF — a Drive link from a Form file-upload question var raw = String(row[48] || '').trim(); if (!raw) return ''; // Case 1: cell already contains a Drive link — pull the file ID straight out of it var idMatch = raw.match(/[-\w]{25,}/); if (idMatch) { return 'https://drive.google.com/file/d/' + idMatch[0] + '/view'; } // Case 2: cell just has a bare filename (older rows) — fall back to matching it // against the fixed Drive folder below var filename = raw.replace(/\.pdf$/i,'').trim().toLowerCase(); var cache = buildPdfCache(); if (cache[filename]) return cache[filename]; var keys = Object.keys(cache); for (var i = 0; i < keys.length; i++) { if (keys[i].indexOf(filename) !== -1 || filename.indexOf(keys[i]) !== -1) return cache[keys[i]]; } return ''; } // Column indices (0-based) in Responses tab // X=23 Y=24 Z=25 AA=26 AB=27 AC=28 AD=29 AE=30 AF=31 AG=32 var EVENT_PAIRS = [ { dateCol: 23, detailCol: 24, label: 'Event 1' }, { dateCol: 25, detailCol: 26, label: 'Event 2' }, { dateCol: 27, detailCol: 28, label: 'Event 3' }, { dateCol: 29, detailCol: 30, label: 'Event 4' }, { dateCol: 31, detailCol: 32, label: 'Event 5' } ]; // ── Parse the combined details column ────────────── // Format: "Event name - XYZ , Makeup by - ABC , Location - Igatpuri , RT - 12 PM" function parseEventDetails(raw) { var result = { eventName: '', makeupBy: '', location: '', readyTime: '' }; if (!raw || raw.trim() === '') return result; // Normalise all RT / Ready Time variants → "RT - " // Handles: RT-, RT -, RT- , Ready Time-, Ready Time -, ready time-, "Ready Time -" etc. raw = raw.replace(/ready\s*time\s*[-–]\s*/gi, 'RT - '); raw = raw.replace(/\bRT\s*[-–]\s*/gi, 'RT - '); // Split on comma — each segment is "Key - Value" var segments = raw.split(','); segments.forEach(function(seg) { var idx = seg.indexOf(' - '); if (idx === -1) return; var key = seg.substring(0, idx).trim().toLowerCase(); var val = seg.substring(idx + 3).trim(); if (!val) return; if (key.indexOf('event name') !== -1 || key === 'event') { result.eventName = val; } else if (key.indexOf('makeup') !== -1 || key.indexOf('mb') !== -1) { result.makeupBy = val; } else if (key.indexOf('location') !== -1 || key === 'loc') { result.location = val; } else if (key === 'rt' || key.indexOf('ready') !== -1) { result.readyTime = val; } }); return result; } // ── Format a date cell value to DD-Mon-YYYY ──────── function formatDate(d) { if (!d) return ''; var dt = (d instanceof Date) ? d : new Date(d); if (isNaN(dt)) return String(d); var months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec']; return dt.getDate() + '-' + months[dt.getMonth()] + '-' + dt.getFullYear(); } // ── Ensure helper sheets exist ───────────────────── function ensureSheets(ss) { var datesSheet = ss.getSheetByName(DATES_SHEET); if (!datesSheet) { datesSheet = ss.insertSheet(DATES_SHEET); datesSheet.getRange(1,1,1,11).setValues([ ['Date','BrideName','EventNo','EventName','Location','ReadyTime','Month','Year','BookingID','PdfUrl','MakeupBy'] ]); } var availSheet = ss.getSheetByName(AVAIL_SHEET); if (!availSheet) { availSheet = ss.insertSheet(AVAIL_SHEET); availSheet.getRange(1,1,1,5).setValues([['Artist','Date','Status','Note','MarkedBy']]); } return { datesSheet: datesSheet, availSheet: availSheet }; } // ── Rebuild BOOKING_DATES from Responses ─────────── function rebuildBookingDates() { var ss = SS_ID ? SpreadsheetApp.openById(SS_ID) : SpreadsheetApp.getActiveSpreadsheet(); var sheets = ensureSheets(ss); var resp = ss.getSheetByName(RESP_SHEET); if (!resp) return; var data = resp.getDataRange().getValues(); // for dates & names var displayData = resp.getDataRange().getDisplayValues(); // for free-text detail cols (keeps RT as "12 PM") // col AW (index 48) = PDF file link from the Form's file-upload question var rows = [['Date','BrideName','EventNo','EventName','Location','ReadyTime','Month','Year','BookingID','PdfUrl','MakeupBy']]; for (var i = 1; i < data.length; i++) { var row = data[i]; var displayRow = displayData[i]; var brideName = String(row[2] || '').trim(); // col C = Client Name if (!brideName) continue; EVENT_PAIRS.forEach(function(pair) { var rawDate = row[pair.dateCol]; var rawDetail = String(displayRow[pair.detailCol] || ''); // ← displayValues: RT stays as "12 PM" if (!rawDate || rawDate === '') return; var parsed = parseEventDetails(rawDetail); var dateStr = formatDate(rawDate); var dt = (rawDate instanceof Date) ? rawDate : new Date(rawDate); // Read PDF URL from column AW (index 48) var pdfUrl = getPdfUrl(row); rows.push([ dateStr, brideName, pair.label, parsed.eventName, parsed.location, parsed.readyTime, isNaN(dt) ? '' : dt.getMonth() + 1, isNaN(dt) ? '' : dt.getFullYear(), String(row[1] || ''), // col B = Booking ID pdfUrl, parsed.makeupBy ]); }); } sheets.datesSheet.clearContents(); sheets.datesSheet.getRange(1, 1, rows.length, 11).setValues(rows); } // ── doGet — returns JSON for the tracker ────────── // Wraps a response as JSONP if a callback param is present (this is what makes // reads reliable cross-browser — script-tag loads aren't subject to CORS) function jsonOut(e, obj) { var cb = e && e.parameter && e.parameter.callback; if (cb) { return ContentService .createTextOutput(cb + '(' + JSON.stringify(obj) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var action = e && e.parameter && e.parameter.action ? e.parameter.action : 'sync'; // ── Return PDF list from Drive folder ────────────── if (action === 'pdfs') { try { var folder = DriveApp.getFolderById('1bClP7P7OoilreV175TMqUBlhdlyH52Gc'); var files = folder.getFiles(); var pdfs = []; while (files.hasNext()) { var f = files.next(); var name = f.getName(); if (name.toLowerCase().indexOf('.pdf') !== -1) { // Make file viewable by anyone with link try { f.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(ex) {} pdfs.push({ name: name.replace(/\.pdf$/i,''), url: 'https://drive.google.com/file/d/' + f.getId() + '/view', id: f.getId() }); } } pdfs.sort(function(a,b){return a.name.localeCompare(b.name);}); return jsonOut(e, {ok:true, pdfs:pdfs}); } catch(err) { return jsonOut(e, {ok:false, error:String(err)}); } } try { var ss = SS_ID ? SpreadsheetApp.openById(SS_ID) : SpreadsheetApp.getActiveSpreadsheet(); ensureSheets(ss); rebuildBookingDates(); // --- Read BOOKING_DATES --- var datesSheet = ss.getSheetByName(DATES_SHEET); var dData = datesSheet.getDataRange().getValues(); var bookings = []; for (var i = 1; i < dData.length; i++) { var r = dData[i]; if (!r[0]) continue; bookings.push({ date: String(r[0]), bride: String(r[1]), eventNo: String(r[2]), eventName: String(r[3]), location: String(r[4]), readyTime: String(r[5]), pdfUrl: String(r[9] || ''), // col J = PDF URL makeupBy: String(r[10] || '') // col K = Makeup by }); } // --- Read Artist Availability --- var availSheet = ss.getSheetByName(AVAIL_SHEET); var aData = availSheet.getDataRange().getValues(); var avail = []; for (var j = 1; j < aData.length; j++) { var a = aData[j]; if (!a[0] || !a[1]) continue; avail.push({ artist: String(a[0]), date: normDate(a[1]), status: String(a[2]), note: String(a[3]) }); } return jsonOut(e, { ok: true, bookings: bookings, avail: avail }); } catch(err) { return jsonOut(e, { ok: false, error: err.toString() }); } } // ── doPost — saves availability markings ────────── // Normalises a date cell to 'yyyy-MM-dd' regardless of whether Sheets stored // it as the original string or auto-converted it to a real Date object. // (Without this, string '2026-07-09' !== Date object, so upsert matching // would silently fail and every save would append a NEW row instead of // updating the existing one — this was the cause of duplicate rows.) function normDate(v) { if (v instanceof Date) { return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } return String(v || ''); } function doPost(e) { try { var ss = SS_ID ? SpreadsheetApp.openById(SS_ID) : SpreadsheetApp.getActiveSpreadsheet(); var availSheet = ss.getSheetByName(AVAIL_SHEET) || ensureSheets(ss).availSheet; var payload = JSON.parse(e.postData.contents); var data = availSheet.getDataRange().getValues(); if (payload.action === 'upsert') { var matchIdx = -1; for (var i = 1; i < data.length; i++) { if (String(data[i][0]) !== payload.artist) continue; if (normDate(data[i][1]) !== payload.date) continue; if (payload.status === 'ALLOT') { // ALLOT rows: match on the SAME event (bride+eventNo prefix) so a // second allotment for a different event on the same day doesn't // overwrite the first — it gets its own row instead. if (payload.eventKey && String(data[i][3] || '').indexOf(payload.eventKey) === 0) { matchIdx = i; break; } } else { // Non-ALLOT statuses (FREE/BUSY/TENTATIVE/LEAVE) stay single-value-per-day if (String(data[i][2]) !== 'ALLOT') { matchIdx = i; break; } } } if (matchIdx >= 0) { availSheet.getRange(matchIdx + 1, 3, 1, 2).setValues([[payload.status, payload.note || '']]); } else { availSheet.appendRow([payload.artist, payload.date, payload.status, payload.note || '', Session.getActiveUser().getEmail()]); } } else if (payload.action === 'delete') { for (var k = 1; k < data.length; k++) { if (String(data[k][0]) !== payload.artist) continue; if (normDate(data[k][1]) !== payload.date) continue; if (payload.eventKey) { // Deleting one specific allotment if (String(data[k][3] || '').indexOf(payload.eventKey) === 0) { availSheet.deleteRow(k + 1); break; } } else { // Deleting a plain (non-ALLOT) status — never touch ALLOT rows here if (String(data[k][2]) !== 'ALLOT') { availSheet.deleteRow(k + 1); break; } } } } return ContentService.createTextOutput('ok'); } catch(err) { return ContentService.createTextOutput('error: '+err.toString()); } }
